\tikzset{%
    every path/.append style={edgeColor},%
    % shapes
    entity/.style={%
        rectangle,%
        minimum size=2.8em%
    },%
    relation/.style={%
        diamond,%
        aspect=2%
    },%
    attribute/.style={%
        ellipse,%
        minimum size=1pt,%
        scale=0.7%
    },%
    matrix/.append style={%
        column sep=0.375in,%
        row sep=1.5in,%
        % default to entity on level 0
        every node/.append style={%
            entity,%
            fill=titleBack,%
            text=titleText%
        },%
        level 1/.append style={%
            every node/.append style={%
                attribute%
            }%
        }%
    },%
    cardinality/.style={%
        postaction={%
            decorate,%
            decoration={%
                raise=-9pt,%
                text along path,%
                text={#1},%
                text color=edgeLabelText,%
                text align={align=center}%
            }%
        }%
    },%
    cardinality/.default={1:1}%
}
